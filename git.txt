// COMMANDES SYSTEME

ls -> lister le répertoire
ls -la -> lister en plus les fichiers/dossiers cachés
cd -> changer de répertoire
cd .. -> revenir en arrière
mkdir -> créer un répertoire
touch -> créer un fichier
pwd -> affiche le chemin d'emplacement
rm (-rf) -> supprimer
clear -> nettoyer

// COMMANDES GIT

git init -> initialise le répertoire GIT
git add . -> ajouter tous les fichiers/dossiers au répertoire
git commit -m "nom" -> créer un commit (une version)
git push -> envoyer cette version
git status -> afficher le statut de la branche actuelle
git clone -> cloner un projet
git reset --hard HEAD -> revenir au dernier commit
git log -> lister les commits
git branch -> lister les branch
git branch nom -> créer une branch
git checkout nom -> changer de branch
